> 1. **简单的反转链表**
>
>    示例：
>
>    输入: 1->2->3->4->5->NULL 
>
>    输出: 5->4->3->2->1->NULL

```
// 方法一 循环
function reverse (head) {
	if (!head) return null
	let pre = null
	let cur = head
	while (cur) {
		let next = cur.next
		cur.next = pre
		pre = cur
		cur = next
	}
	return pre
}

// 方法二 递归
function reverse (head) {
	if (!head) return null
	let help = (pre, cur) => {
		if (!cur) return pre
		let next = cur.next
		cur.next = pre
		return help(cur, next)
	}
	return help(null, head)
}
```

> 4. 删除重复节点

```javascript
function deleteNode (head) {
    if (!head || !head.next) {
        return head
    }
    let dummy = new ListNode()
    dummy.next = head
    let cur = dummy
    while(cur.next && cur.next.next) {
        if (cur.next.val === cur.next.next.val) {
            // 记录当前值 排查后面有多个相同的节点值的情况
            let val = cur.next.val
            while(cur.next && cur.next.val === val) {
                cur.next = cur.next.next
            }
        } else {
            cur = cur.next
        }
    }
    return dummy.next
}
```

```javascript
function reoveNthFromEnd (head, n) {

    let slow = dummy
```

```javascript
function loop (head) {
    let slow = head
    let fast = head
    while (fast && fast.next) {
        slow = slow.next
        fast = fast.next.next
```

> 7. 找到链表的环节点

```javascript
// 快慢指针找环节点
// 思路: 假设 slow 指针和 fast 指针在离环节点 y 处相遇，此时假设 slow 指针走了 x 步，则有 x - L = m * S + y
// 对于 fast 指针来说 2x - L = n * S + y
// 两个公式结合得到 L = (n - m) * S - y
// 假设有一个新指针从头节点出发和 slow 指针同时走L步，此时新指针到达环节点，slow 指针走了 L = （n-m）* S - y 步, 原来相对环
// 节点来说 slow指针 此时的位置在 y + (n - m) * S - y = (n - m) * S 位置，即在环节点的位置，因此和新指针相遇。
function findNode (head) {
    let dummy = new ListNode ()
    let slow = dummy
    let fast = dummy
    while (fast && fast.next) {
        fast = fast.next.next
        slow = slow.next
        if (slow === fast) {
            // 相遇后让新节点从头节点位置和slow 指针同时走
            let p = dummy
            while(p!==slow) {
                p = p.next
                slow = slow.next
            }
            return p
        }
    }
    return null
}
```
