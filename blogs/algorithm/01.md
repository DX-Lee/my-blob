**优先队列**

堆本质上是一种二叉树，但是这棵二叉树比较特殊，除了用数组来依次存储各个节点(节点对应的数组下标和**层序遍历的序号**一致)之外，它需要保证**任何一个父节点的优先级大于子节点**，这也是它最关键的性质，因为保证了根元素一定是优先级最高的。

```
先实现一个最大堆
clss MaxHeap {
	constructor (arr, compare) {
		this.data = arr
		this.size = arr.length
		this.compare = compare
	}
	getSize () {
		return this.size
	}
	isEmpty () {
		return this.size === 0
	}
	add (value) {
		this.data.push(value)
		this.size++
		this._siftUp(this.getSize() - 1)
	}
	findMax () {
		if (this.getSize() === 0) {
			return
		}
		return this.data[0]
	}
	// 让优先级最大的元素出队列 即队首元素
	extractMax () {
		// 先要保存队首元素
		let ret = this.data[0]
		// 将队首元素与队尾元素交换位置
		this._swap(0, this.getSize() - 1)
		// 将对首元素踢出去
		this.data.pop()
		this.size--
		// 对新对首 sighDown
		this._siftDown(0)
		return ret
	}
	toString () {
		console.log(this.data)
	}
	_swap(i, j) {
		[this.data[i], this.data[j]] = [this.data[j], this.data[i]]
	}
	_parent (index) {
		return Math.floor((index - 1) / 2)
	}
	_leftchild (index) {
		return 2 * index + 1
	}
	_rightchild(index) {
		return 2 * index + 2
	}
	_siftUp(k){
	// 只要 子元素优先级大于父元素就上浮
		while (k > 0 && this.compare(this.data[k], this.data[this._parent(k)])) {
			this._swap(k, this._parent(k))
			k = this._parent(k)
		}
	}
	_siftDown (k) {
		// 当存在左孩子的时候
		while(this._leftchild(k) < this.size) {
			let j = this._leftchild(k)
			// 当存在右孩子 且 右孩子大于左孩子时候
			if(this._rightchild(k) && this.compare(this.data[this._rightchild(k)], this.data[j])) {
				j++
			}
			if (this.compare(this.data[k], this.data[j])) {
				// 父节点大于子节点 不处理
				return
			}
			// 否则交换位置 继续下沉
			this._swap(k, j)
			k = j
		}
	}
}
```
